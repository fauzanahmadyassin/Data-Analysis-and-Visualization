Data scraping process¶

In [1]:
i=1
url = "https://www.mudah.my/Malaysia/Mobile-Phones-and-Gadgets-3020/iphone-all?o="+str(i)+"&q=iphone&th=1"

while (i<200):

    page = requests.get(url)
    soup = BeautifulSoup(page.content, 'lxml')

    phone_name = soup.find_all('h2',{'class':'list_title'})
    phone_price = soup.find_all('div',{'class':'ads_price'})
    area = soup.find_all('div',{'class':'area'})
    condition = soup.find_all('font',{'class':'icon_label'})
    seller = soup.find_all('div',{'class':'container_ico'})
    
    
    phone_name = [a.text for a in phone_name]
    phone_name = phone_name[:-1]
    
    phone_price = [a.text for a in phone_price]
    phone_price = phone_price[:-1]
    
    area = [a.text for a in area]
    area = area[:-1]
    
    condition = [a.text for a in condition]
    condition = condition[:-1]
    
    seller = [a.text for a in seller]
    seller = seller[:-1]
   

    list_name = list_name + phone_name
    list_price = list_price + phone_price
    list_area = list_area + area
    list_cond = list_cond + condition
    list_seller = list_seller + seller
    
    print ('Data from page '+str(i)+' has been scraped')
    
    i = i+1    
    url = "https://www.mudah.my/Malaysia/Mobile-Phones-and-Gadgets-3020/iphone-all?o="+str(i)+"&q=iphone&th=1"
    #print (url)
